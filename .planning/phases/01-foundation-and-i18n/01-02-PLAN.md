   ```typescript

   import '@/styles/globals.css';

         <body className={`${inter.variable} font-sans antialiased`}>
           {children}
         </body>
       </html>
     );
   }
   ```

3. Update `src/app/[locale]/layout.tsx` to wrap children with ThemeProvider from next-themes:
   - Add ThemeProvider inside NextIntlClientProvider
   - Set `attribute="class"`, `defaultTheme="system"`, `enableSystem={true}`, `disableTransitionOnChange`
   - Keep all existing i18n logic (generateStaticParams, generateMetadata, setRequestLocale, getMessages)
   ```typescript
   import { ThemeProvider } from 'next-themes';
   // ... existing imports ...

   export default async function LocaleLayout({ children, params }) {
     // ... existing locale validation and message loading ...

     return (
       <NextIntlClientProvider messages={messages}>
         <ThemeProvider
           attribute="class"
           defaultTheme="system"
           enableSystem
           disableTransitionOnChange
         >
           {children}
         </ThemeProvider>
       </NextIntlClientProvider>
     );
   }
   ```

4. Create `src/components/layout/theme-toggle.tsx`:
   ```typescript
   'use client';

   import { Moon, Sun } from 'lucide-react';
   import { useTheme } from 'next-themes';
   import { Button } from '@/components/ui/button';

   export function ThemeToggle() {
     const { setTheme, theme, resolvedTheme } = useTheme();

     return (
       <Button
         variant="ghost"
         size="icon"
         onClick={() => setTheme(resolvedTheme === 'dark' ? 'light' : 'dark')}
         className="relative"
         aria-label="Toggle theme"
       >
         <Sun className="h-5 w-5 rotate-0 scale-100 transition-all dark:-rotate-90 dark:scale-0" />
         <Moon className="absolute h-5 w-5 rotate-90 scale-0 transition-all dark:rotate-0 dark:scale-100" />
         <span className="sr-only">Toggle theme</span>
       </Button>
     );
   }
   ```
