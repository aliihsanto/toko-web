const BASE_URL = 'https://toko.com.tr';

function makeEntry(path, priority, changeFrequency): MetadataRoute.Sitemap[number] {
  return {
    url: `${BASE_URL}/tr${path}`,
    lastModified: new Date(),
    changeFrequency,
    priority,
    alternates: {
      languages: Object.fromEntries(
        routing.locales.map((locale) => [locale, `${BASE_URL}/${locale}${path}`])
      ),
    },
  };
}

return [

**Priority rationale:**
- Product import pages: 0.7 (high commercial value, similar to blog posts)
- Country trade guides: 0.7 (high commercial value)
- FAQ pages: 0.6 (informational, lower conversion intent)
- Customs pages: 0.6 (informational, lower conversion intent)
